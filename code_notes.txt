ya da tek tek komut satırında sıralamak yerine bir excel listesi vermek süretiyle de yapılabilir. aşağıda bir excel dosyası içindeki
cik ve ref-years'a göre çalıştırılmıi bir komut bulunmaktadır. Yenilik olarak

--path excel dosyasnının yolu. (xlsx ve csv formatını desteklemektedir.
--id-col excel dosyasındaki cik kodlarının bulunduğu columun adı.
--year-col excel dosyasındaki years kodlarının bulunduğu columun adı.

edgar-fetcher build-master \
  --path "/Users/emre/Downloads/Relocations08062025.xlsx" \
  --id-col "cik" \
  --year-col "EffectiveYear" \
  --forms all \
  --groups core \
  --download "/Users/emre/Documents/PYprojects/" \
  --basis filed \
  --out "/Users/emre/Documents/PYprojects/master_index.csv" \
  --name "Emre Topalgokceli" \
  --institution "Indiana University" \
  --email "etopa@iu.edu"
